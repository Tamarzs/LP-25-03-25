1. repetição com for

/* inicialização */
int contador = 1;
int fim = 5;

/* condição */
while( contador <= fim ) {
  printf("%d ", contador);

  /* atualização */
  contador = contador + 1;
}

No comando for todos os itens da repetição (inicialização, condição e atualização) são definidos de uma vez: 

int fim = 5;

for(int contador = 1; contador <= fim; contador++) {
  printf("%d ", contador);
}

O resultado é o mesmo!

2. Switch

Se estiver usando um único valor de comparação em vários if você pode usar o comando switch. 

#include <stdio.h>

int main(void) {
  char tamanho;

  printf("Digite o tamanho recebido, (F sair):");
  scanf("%c", &tamanho);

  int quantidadeP = 0;
  int quantidadeM = 0;
  int quantidadeG = 0;

  while( tamanho != 'F' ) {
    if( tamanho == 'P' ) {
      quantidadeP++;
    }
    if( tamanho == 'M' ) {
      quantidadeM++;
    }
    if( tamanho == 'G' ) {
      quantidadeG++;
    }

    printf("Digite o tamanho recebido, F para encerrar:");
    scanf("%c", &tamanho);
  }

  printf("P: %d M: %d G: %d", quantidadeP, quantidadeM, quantidadeG);
  return 0;
}

Podemos usar switch assim: 

#include <stdio.h>

int main(void) {
  char tamanho;

  printf("Digite o tamanho recebido, F para encerrar:");
  scanf("%c", &tamanho);

  int quantidadeP = 0;
  int quantidadeM = 0;
  int quantidadeG = 0;

  while( tamanho != 'F' ) {
    switch (tamanho) {
      case 'P':
	quantidadeP++;
	break;
      case 'M': 
	quantidadeM++;
	break;
      case 'G':
	quantidadeG++;
	break;
    }

    printf("Digite o tamanho recebido, F para encerrar:");
    scanf("%c", &tamanho);
  }

  printf("P: %d M: %d G: %d", quantidadeP, quantidadeM, quantidadeG);
  return 0;
}

Sem o break, a execução continua nos outros casos: 

while( tamanho != 'F' ) {
  switch (tamanho) {
    case 'P':
      quantidadeP++;	
    case 'M': 
      quantidadeM++;	
    case 'G':
      quantidadeG++;	
  }

Podemos receber uma tecla apertada usando getchar(): 

char letra;

printf("Aperte uma tecla: ");

letra = getchar();

printf("Tecla apertada: %c", letra);

2.2. Caso padrão

O caso padrão default executa se nenhum case for encontrado: 

#include <stdio.h>

int main(void) {
  char tamanho;

  printf("Digite o tamanho recebido, F para encerrar:");
  scanf("%c", &tamanho);

  int quantidadeP = 0;
  int quantidadeM = 0;
  int quantidadeG = 0;

  while( tamanho != 'F' ) {
    switch (tamanho) {
      case 'P':
	quantidadeP++;
	break;
      case 'M': 
	quantidadeM++;
	break;
      case 'G':
	quantidadeG++;
	break;
      default:
	printf("Tamanho não encontrado!");
	break;
    }

    printf("Digite o tamanho recebido, F para encerrar:");
    scanf("%c", &tamanho);
  }

  printf("P: %d M: %d G: %d", quantidadeP, quantidadeM, quantidadeG);
  return 0;
}

2.3. Ignorar linha em branco

Para evitar a repetição do prompt ao usuário basta ignorar a tecla ENTER digitada usando: 

case '\n':
case '\t':
case '\r':
case ' ':
  break;
